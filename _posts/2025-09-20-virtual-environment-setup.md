# 가상환경 설정

## 가상환경 생성
pandas 폴더에서 **git bash** 열기  
python -m venv venv 입력

## 가상환경 활성화
source venv/Scripts/activate 입력

## 가상환경에서의 도구 설치
- pip install jupyterlab 
- pip install openpyxl
- pip install pandas
- pip install numpy
- pip install seaborn

## 가상환경 종료
deactivate 입력
